CREATE TABLE tp_dbc_cohort_table AS
SELECT
* FROM (
            SELECT
  res.dr_screening.cohort_definition_id,
  res.dr_screening.subject_id,
  res.dr_screening.cohort_start_date,
  res.dr_screening.cohort_end_date,
  EXTRACT(YEAR FROM res.dr_screening.cohort_start_date) - dbo.person.year_of_birth AS age,
  dbo.concept.concept_name AS sex,
  subject_id_origin
FROM
  res.dr_screening
INNER JOIN dbo.person
  ON res.dr_screening.subject_id = dbo.person.person_id
INNER JOIN dbo.concept
  ON dbo.person.gender_concept_id = dbo.concept.concept_id
INNER JOIN
  (
    SELECT
      ROW_NUMBER() OVER (PARTITION BY subject_id ORDER BY subject_id) AS subject_id,
      CAST(subject_id AS VARCHAR) AS subject_id_origin
    FROM res.dr_screening
    GROUP BY subject_id
  ) org_subject_table
  ON subject_id_origin = res.dr_screening.subject_id
WHERE
